#   RUNS_FILE=experiments/navier_stokes/ono_mean_latent_engineered_data.txt bash scripts/hpc/runner.sh

# Output layout:
#   outputs/navier_stokes/mean_constraint/ono/<budget>/seed_42/{resolved_config.yaml,best.pt}
